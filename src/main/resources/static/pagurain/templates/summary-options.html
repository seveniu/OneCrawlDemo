$p.sizer(Seletor, Options,[SelectDatas]);

Options

参数 默认值 数据类型 描述
cName cName string 数据源data cName的key
allColumns -- array 所有的列，数据格式: 
{
        //dataName
        "dataName": "title",
        //是否为Title字段 默认填充列名
        "isTitle": true,
        //自定义class
        "klass": "summary-span-title",
        "tpl": "增长数:{0}" //模板填充,
        //key为datame的值 填充方法(优先级高于tpl)
        'render': function (data, full) {}
}
allRows -- array 所有的行,数据格式: 
{
    //cName key默认为cName可在option中配置
    "cName": "showCounts",
    //title
    "title": "展示次数"
}
[saveState] false boolean 保存状态到cookies
[maxNum] 最大列数 int 最大展示数
[minNum] 1 int 最小展示数
[canChoose] false boolean 是否可以切换面板，如果为true则点击后有选中效果
[showSetting] false boolean 是否显示设置按钮
[dataSource] -- function 一个方法，定义dataListview显示的数据源
[dataParams] {} object 设置自定义请求参数
[callBackPanel] -- function(columnName,columnData,allDatas){} 点击面板的回调
columnName点击的列名
columnData当前列数据
allDatas所有数据
[callBackGetData] -- function 数据渲染后的回调
 [canChoose=true]
"callBackGetData":function(_chooseColumnsName, _chooseColumnsData, _allDatas){}  //选中的列名,选中列的数据,所有数据
 [canChoose=false]
"callBackGetData":function( _allDatas){}  //所有数据

Options(showSetting为true时有效)

参数 默认值 数据类型 描述
[callbackOpen] -- function 展开面板的回调
[callbackSubmit] -- function(showColumns,datas){} 点击确定按钮的回调
showColumns:展示的列
datas:全部数据
[callbackCancel] -- function 点击取消按钮的回调
