Functions

/**
* 获取配置
* @return {Object}
**/
Sizer.getOption();
/**
*获取全部数据
*@return {Array}
**/
Sizer.getAllDatas();
/**
* 获取已选数据
* @return {Array}
**/
Sizer.getSelectDatas();
/**
* 选中数据
* @param values {array|string|obj}
* array:[{name:"2.1.12.3",value:1},...]
* obj:{name:"2.1.12.3",value:1},
* string:"all"全选,"null"全不选 ,如果为单选则接受直接传入value进行选中
* return sizer
*/
Sizer.chooseData(data);
/**
* 动态绑定事件
* @param functionName {String}
* expand,close,option,search,clean,submit,cancel
* @param callback {Function}
* @return Sizer;
**/
Sizer.on(functionName,callback);
/**
* 取消事件绑定
* @param functionName {String}
* expand,close,option,search,clean,submit,cancel
* @return Sizer;
**/
Sizer.off(functionName);
/**
* 销毁插件
*/
Sizer.destroy();
/**
* 切换面板状态
*/
Sizer.expandPanel();
/**
* 打开面板
*/
Sizer.open();
/**
* 关闭面板
*/
Sizer.close();
/**
* 更新数据
* @param params{Object}
*/
Sizer.update(params);

Options

参数 默认值 数据类型 描述
[isMultiple] false boolean 是否为多选
[isExpand] false boolean 是否展开
[btns] 多选:['selectAll','clearAll'],单选:['clearSingle'] array sizer提供的按钮默认为全部提供
dataSource -- function 定义dataListview显示的数据源的方法
dataParams -- object|function 设置自定义请求参数
[dataMapping] {
"name": "name",
"value": "id"
}		 object 数据映射
[position] {
left: 0
}		 object 面板的相对位置(以按钮为参照物)
[style] -- string 自定义class
[processing] <i class="fa fa-spinner fa-spin big"></i>  加载中 string loading默认文字
[search] 请输入查询字段 string 搜索框默认文字
[callbackExpand] -- function 面板展开时的回调
[callbackClose] -- function([datas,allDatas]){} 面板关闭时的回调
datas:选中的数据
allDatas:全部数据
[callbackOption] -- function([data]){} 点击选项时的回调
data:选中的数据
[callbackSearch] -- function([data]){} 点击选项时的回调
data:搜索到的数据
[callbackClean] -- function([data]){} 点击清除按钮时的回调
data:选择的数据
[callbackLoadData] -- function([data,data]){} 点击清除按钮时的回调
data:选择的数据

Options(isMultiple为true时有效)

参数 默认值 数据类型 描述
[callbackSelectAll] -- function([data]){} 搜索框输入时的回调
data:符合搜索条件的数据
[callbackSubmit] -- function([selectData,allData]){} 点击确定按钮时的回调
selectData:选中数据
allData:全部数据
[callbackCancel] -- function(){} 点击取消按钮时的回调

Options.dataMapping

参数 默认值 数据类型 描述
name name string 选项显示的字段
value value string 选项的值

SelectDatas

参数 默认值 数据类型 描述
-- -- array 数据格式如下: 
[{
  "id": 4 //选中项的id
  "name": "2.1.4" //此数据用于设置默认的选中文字，如需提示文字，则不需要此设置 仅在isMultiple为false时有效
},
...
]或者
"all","null"
如果name不为空则使用name的值覆盖原默认提示文字
